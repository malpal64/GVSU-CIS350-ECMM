16   
17       def update(self, player):
18           shot_cooldown = 300
19           arrow = None
20   
21           self.rect.center = player.rect.center
22   
23           pos = pygame.mouse.get_pos()
24           x_dist = pos[0] - self.rect.centerx
25           y_dist = -(pos[1] - self.rect.centery) # y-coord increase down screen
26           self.angle = math.degrees(math.atan2(y_dist, x_dist))
27   
28           # get mouse click
29           if pygame.mouse.get_pressed()[0] and self.fired == False and (pygame.time.get_ticks() - self.last_shot) >= shot_cooldown:
30               arrow = Arrow(self.arrow_image, self.rect.centerx, self.rect.centery, self.angle)
31               self.fired = True
32               self.last_shot = pygame.time.get_ticks()
33           # reset mouseclick
34           if pygame.mouse.get_pressed()[0] == False:
35               self.fired = False
36           return arrow
37   
38       def draw(self, surface):
39           self.image = pygame.transform.rotate(self.original_image, self.angle)
40           surface.blit(self.image, ((self.rect.centerx - int(self.image.get_width()/2)), self.rect.centery - int(self.image.get_height()/2)))
